vue双向绑定

发表于 2018-04-24

实现一个简单的vue双向绑定，text内容跟随input输入框的内容变化

上次理解了vue响应式原理，通过设置访问属性，set读取属性值调用，get设置属性值调用，从而实现对对象属性的监听。vue是一个mvvm，Model-View-ViewModel，传统MVC模式，更新model后，view也会跟着更新，但双向绑定，当view更新，model也会更新。所以最直接的就是用户填写表单，view更新可以自动更新model，从而实现简单的双向绑定

function node2Fragment(node, vm) {
  // 虚拟文档仓库，保存将来可能会添加到文档中的节点
  var flag = document.createDocumentFragment()
  var child
  while (child = node.firstChild) {
      // 编译节点
      compile(child, vm)
      // 如果传入的节点已经是文档的一部分，那会将该节点从原来的位置转移到新位置上
      flag.appendChild(child)
    }
    return flag
  }

  function compile(node, vm) {
    // 双大括号正则匹配
    var reg=/\{\{(.*)\}\}/
    // 节点类型为element
    if(node.nodeType === 1) {
      var attr = node.attributes
      for(let i = 0; i < attr.length; i++) {
      if(attr[i].nodeName === 'v-model') {
        var name = attr[i].nodeValue
                      node.addEventListener('input', function (e) {
                          vm[name] = e.target.value
                      })
                      // 值未发生改变时，使用初始值
                      node.value = vm[name] || vm.data[name]
                  }
              }
          }
          // 节点类型为text
          if(node.nodeType === 3) {
              if(reg.test(node.nodeValue)) {
                  //与正则表达式匹配的第一个 子匹配
                  var name = RegExp.$1
                  // 去掉空格
                  name = name.trim()
                  // node.nodeValue = vm.data[name]
                  //初始化数据，并给对应的data属性值添加观察者
                  new Watcher(vm, node, name)
              }
          }
      }
      function Watcher(vm, node, name) {
          //Dep.target是一个Dep的静态属性,表示当前观察者。
          Dep.target = this
          this.name = name
          this.node = node
          this.vm = vm
          // 订阅者执行一次更新视图
          this.update()
          Dep.target = null
      }

      Watcher.prototype = {
          update: function () {
              //触发对应属性值得get函数
              this.get()
              this.node.nodeValue = this.value
          },
          get: function () {
              this.value = this.vm[this.name]
          }
      }

      function defineReactive(obj, key, val) {
          //定义一个主题
          var dep = new Dep()
          Object.defineProperty(obj, key, {
              get: () => {
                   //添加订阅者watcher到主题对象Dep
                  if(Dep.target) dep.addSub(Dep.target)
                  return val
              },
              set: newVal => {
                  // 如果值不变，return
                  if(val === newVal) {
                      return 
                  }
                  val = newVal
                  //作为发布者发出通知（更新所有订阅了这个属性的view）
                  dep.notify()
                  console.log(`你设置了${key}的值为${newVal}`)
              }
          })
      }

      function Dep() {
          // 主题的订阅者
          this.subs = []
      }

      Dep.prototype = {
          //添加订阅者方法
          addSub: function (sub) {
              this.subs.push(sub)
          },
          // 发布信息的方法（让订阅者更新视图view）
          notify: function () {
              this.subs.forEach(v => v.update())
          }
      }

      // 观察属性的变化
      function observe(obj, vm) {
          Object.keys(obj).forEach((key) => {
              if (obj.hasOwnProperty(key)) {
                  this.defineReactive(vm, key, obj[key]);
              }
          })
      }
      // vue对象
      function Vue(options) {
          this.data = options.data
          var id = options.el
          observe(this.data, this)
          var dom = node2Fragment(document.getElementById(id), this)
          document.getElementById(id).appendChild(dom)
      }

DOM知识点

  • document.createDocumentFragment()创建了一虚拟的节点对象，节点对象包含所有属性和方法，虚拟DOM
  • nodeType返回元素的节点类型
  • nodeValue设置或返回元素值
  • firstNode返回元素的第一个子元素
  • attributes返回元素属性的 NamedNodeMap
